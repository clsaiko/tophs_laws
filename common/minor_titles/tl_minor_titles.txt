# MINOR TITLES
# Characters can have any number of these
#
#	Titles:
#		Tresviri capitales - oversee imperial executions
#		Tresviri epulones - priests that oversee public banquets
#		Tresviri monetales - oversee imperial coin minting
#		Senator titles - Duke level titles to add further vassals to the council
#		Legatus Augusti titles - King level titles to add further vassals to the council
#		
#		keeper of the seal
#		
#
########################################################
#	grant_limit: Max number of holders.
#	realm_in_name: Append the name of the country after the title (Queen [of Sweden])
#	allowed_to_hold: conditions for the character to hold the title
#	allowed_to_grant: conditions for a ruler to grant the title
#	regent: This is the title for regents (not really "minor")
#	gain_effect: fires when the title is granted
#	lose_effect: fires when a title is lost if dismissal_trigger evaluates to false
#	message: should fire a message to the its owner when set
#	The other fields are all character modifiers.

# Basic title example
title_houndmaster = {
    dignity = 0.05
    grant_limit = 1
    opinion_effect = 5

    monthly_salary = 0.1
    monthly_prestige = 0.15

    is_unique = yes

    allowed_to_hold = {
        is_adult = yes
        NOT = { trait = incapable }
    }
    allowed_to_grant = {
		is_feudal = yes
		OR = {
			has_landed_title = e_byzantium
			has_landed_title = e_roman_empire
		}
    }

	revoke_trigger = {
		FROM = {
			primary_title = {
				NOT = { 
					AND = {
						ROOT = { is_powerful_vassal = yes }
					}
				}
			}
		}
	}
}

# Tresviri monetales - oversee imperial coin minting
tbyz_title_monetalis = {
	is_voter = no
	attribute = stewardship	#used in minor titles?
	dignity = 0.1
	realm_in_name = yes
	grant_limit = 3
	opinion_effect = 10
	
	monthly_salary = 0.15
	monthly_prestige = 0.25
	
	show_as_title = yes
	
	is_unique = yes
	
	allowed_to_hold = {
		NOT = { 
			primary_title = { temporary = yes } 
		}
		is_landed = yes
		is_adult = yes
		NOT = { has_trait = incapable }
		NOT = { has_trait = infirm }
		NOT = { has_trait = imbecile }
		higher_tier_than = BARON
		NOT = { trait = incapable }
	}
	
	allowed_to_grant = {
		is_feudal = yes
		OR = {
			has_landed_title = e_byzantium
			has_landed_title = e_roman_empire
		}
		OR = {
			religion_group = christian
			religion_group = jewish_group
			religion_group = pagan_group
		}
	}
	
	gain_effect = {
	}
	lose_effect = {
	}

	message = yes
}

# Imperial Senate - Consul
tbyz_title_consul = {
	is_voter = yes
	is_high_prio = yes
	attribute = diplomacy	#used in minor titles?
	dignity = 5
	realm_in_name = yes
	grant_limit = 1
	opinion_effect = 10
	
	monthly_salary = 0.0	#unpaid
	monthly_prestige = 0.5
	
	show_as_title = yes
	
	is_unique = no
	
	allowed_to_hold = {
		NOT = { 
			primary_title = { temporary = yes }
		}
		#primary_title = { is_vice_royalty = yes }
		OR = {
			NOT = { is_female = yes }
			OR = {
				liege = { has_law = status_of_women_3 }
				liege = { has_law = status_of_women_4 }
			}
		}

		NOT = { is_republic = yes }
		NOT = { is_theocracy = yes }
		OR = {
			is_adult = yes
			liege = { has_trait = arbitrary }
			liege = { has_trait = lunatic }
		}
		NOT = { has_trait = incapable }
		NOT = { has_trait = infirm }
		NOT = { has_trait = imbecile }
		NOT = { is_councillor = yes }	#can't already be a councillor
	}
	
	allowed_to_grant = {
		is_feudal = yes
		has_law = tl_senate_law_1
		OR = {
			has_landed_title = e_byzantium
			has_landed_title = e_roman_empire
		}
		OR = {
			religion_group = christian
			religion_group = jewish_group
			religion_group = pagan_group
		}
	}
	
	gain_effect = {
	}
	lose_effect = {
	}

	message = yes
}#end tbyz_title_consul
